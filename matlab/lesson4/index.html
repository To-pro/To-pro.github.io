結合前面的介紹，我們可以將無人機飛行模擬結果繪製成動畫。

範例中的 MAT 檔 (auto_fly_90.mat) 裡面有兩個結構變數分別是飛行資料 data 和期望軌跡 desired_trajectory

  • data:

    • t: 時間
    • x: 無人機 x 位置
    • y: 無人機 y 位置
    • z: 無人機 z 位置
    • P:[x;y;z] 無人機位置
    • phi,theta,psi: 無人機姿態

  • desired_trajectory:

    • waypoint: 航點
    • x_d: 無人機期望 x 位置
    • y_d: 無人機期望 y 位置
    • z_d: 無人機期望 z 位置
    • phi_d,theta_d,psi_d: 無人機期望姿態
    • p: 小圖期望位置

範例的 STL 與 MAT 可以在這邊下載: 載點

# 程式碼:

%全地圖畫法
clc;clear;close all;

addpath 'Z:\My Drive\MATLAB\plot_function'
addpath(genpath('.\STL\450'))

load auto_fly_90.mat;

video_name='auto_fly_90';

%使用者設定開始

fuselage_color = [0.9,0.9,0.8];
propeller_color = [0 0 0];
data_color = 'b';
stl_amplification = 0.5;

reference_color = 'g';
waypoint_color = 'r';
time_window = 8;%sec

%使用者設定結束

%% 繪製動畫準備
video=figure();
set(video,'Position',[1 49 1.7067e+03 946],'color','w');%需放大錄製，不然畫質很差

subplot('Position', [0.06,0.035,0.7,0.95]);
g_main = gca;
hold on

%透明球(避免邊界隨螺旋槳變化)
[bal.x,bal.y,bal.z] = sphere;
bal.x = bal.x*0.5*stl_amplification;
bal.y = bal.y*0.5*stl_amplification;
bal.z = bal.z*0.5*stl_amplification;
ball = surf(bal.x,bal.y,bal.z,"FaceAlpha",0,"LineStyle","none");


%機架
F450_raw = stlread("F450_no_propeller.stl");
F450_poinit = stl_amplification*F450_raw.Points;           %取點
F450_conn = F450_raw.ConnectivityList;   %取三角形
clear F450_raw;
F450_triang = triangulation(F450_conn,F450_poinit);
F450 = trisurf(F450_triang,'facecolor',fuselage_color,'edgecolor','none');
hold on

%槳葉
propeller_raw = stlread("propeller.stl");
propeller_poinit = stl_amplification*propeller_raw.Points;           %取點
propeller_conn = propeller_raw.ConnectivityList;   %取三角形
clear propeller_raw;
propeller_triang = triangulation(propeller_conn,propeller_poinit);

for i=1:4
    propeller{i} = trisurf(propeller_triang,'facecolor',propeller_color,'edgecolor','none');
end
propeller_RT_B_to_motor = cell(4,1);
propeller_deg = [pi*0.1;pi*0.2;pi*0.3;pi*0.4];



%% 開燈
camlight("headlight")


%% 文字設定
digits_set='%5.2f';

Time_txt=text('String',['time: ', '00.00' ,' s'],...
              'Interpreter','latex',...
              'FontSize',30,...
              'Units','normalized',...
              'Position',[ 0.05 0.95 0]);

% t = data.t(1:3000);
t = data.t;
dt = t(2)-t(1);

%% 初始狀態
%透明球
ball.XData = bal.x + data.P(1,1);
ball.YData = bal.y + data.P(2,1);
ball.ZData = bal.z + data.P(3,1);

%機架
F450_dcm = angle2dcm (data.psi(1),data.theta(1),data.phi(1));
F450_RT = [F450_dcm',data.P(:,1);[0,0,0,1]];
F450_F450_poinit_new  = (F450_RT*[F450_poinit';ones(1,length(F450_poinit))])';
F450_F450_poinit_new=F450_F450_poinit_new(:,1:3);
F450.Vertices = F450_F450_poinit_new;

%槳葉
propeller_offset{1} = [ 159.1;  159.1; 196.7500]*0.001*stl_amplification;
propeller_offset{2} = [-159.1;  159.1; 196.7500]*0.001*stl_amplification;
propeller_offset{3} = [-159.1; -159.1; 196.7500]*0.001*stl_amplification;
propeller_offset{4} = [ 159.1; -159.1; 196.7500]*0.001*stl_amplification;



for i=1:4
    propeller_RT_B_to_motor{i} = [angle2dcm(propeller_deg(i),0,0)' propeller_offset{i};[0,0,0,1]];
    propeller_F450_poinit_new  = (F450_RT*propeller_RT_B_to_motor{i}*[propeller_poinit';ones(1,length(propeller_poinit))])';
    propeller_F450_poinit_new=propeller_F450_poinit_new(:,1:3);
    propeller{i}.Vertices = propeller_F450_poinit_new;
end


%% 全部的軌跡
data_Trajectory_point=[ones(1,length(t))*data.P(1,1);ones(1,length(t))*data.P(2,1);ones(1,length(t))*data.P(3,1)];
data_Trajectory=plot3(data_Trajectory_point(1,:),data_Trajectory_point(2,:),data_Trajectory_point(3,:), ...
    '-o','MarkerSize',5,'MarkerFaceColor',data_color ...
    ,'MarkerEdgeColor','none',"LineWidth",2,'Color',data_color,'MarkerIndices',1:10:length(t));



%% 航點與期望軌跡

hold on
p(2)=plot3(desired_trajectory.p(1,:),desired_trajectory.p(2,:),desired_trajectory.p(3,:),'-',"Color",reference_color,"LineWidth",1.5);
hold on
plot3(desired_trajectory.waypoint(1,1:end),desired_trajectory.waypoint(2,1:end),desired_trajectory.waypoint(3,1:end),"Color",waypoint_color,'LineStyle','-.',"LineWidth",1.5);
hold on
plot3(desired_trajectory.waypoint(1,1),desired_trajectory.waypoint(2,1),desired_trajectory.waypoint(3,1),'o',"Color",reference_color,'LineStyle','none',"LineWidth",2);
hold on
p(1)=plot3(desired_trajectory.waypoint(1,2:end),desired_trajectory.waypoint(2,2:end),desired_trajectory.waypoint(3,2:end),'o',"Color",waypoint_color,'LineStyle','none',"LineWidth",2);

legend([p data_Trajectory],'Waypoint','Desired trajectory','Cascade PID','Location',[0.5955 0.9003 0.1107 0.0751])

xlabel("X(m)");
ylabel("Y(m)");
zlabel("Z(m)");


% axis equal;
axis padded
grid on;
view(3)
plot_set(16);


writerObj = VideoWriter(video_name,'MPEG-4');
writerObj.FrameRate = 25;
writerObj.Quality = 100;
open(writerObj);
n = length(t);


%% 其他子圖
RAD2DEG = 180/pi; 
p_y = [data.x;
       data.y;
       data.z;
       RAD2DEG*data.phi;
       RAD2DEG*data.theta;
       RAD2DEG*data.psi];

subplot("Position",[0.78 0.82 0.2 0.13]);
p(1) = plot(t,p_y(1,:),"Color",data_color,"LineStyle","-");
hold on
plot(t,desired_trajectory.p(1,:),"LineStyle","--","Color",reference_color);
set(gca,'xticklabel',[])
g(1) = gca;
ylabel("$x$ (m)","Interpreter","latex");
plot_set(16,2);

subplot("Position",[0.78 0.67 0.2 0.13]);
p(2) = plot(t,p_y(2,:),"Color",data_color,"LineStyle","-");
hold on
plot(t,desired_trajectory.p(2,:),"LineStyle","--","Color",reference_color);
set(gca,'xticklabel',[])
g(2) = gca;
ylabel("$y$ (m)","Interpreter","latex");
plot_set(16,2);

subplot("Position",[0.78 0.52 0.2 0.13]);
p(3) = plot(t,p_y(3,:),"Color",data_color,"LineStyle","-");
hold on
plot(t,desired_trajectory.p(3,:),"LineStyle","--","Color",reference_color);
set(gca,'xticklabel',[])
g(3) = gca;
ylabel("$z$ (m)","Interpreter","latex");
plot_set(16,2);

subplot("Position",[0.78 0.37 0.2 0.13]);
p(4) = plot(t,p_y(4,:),'Color',data_color,"LineStyle","-");
set(gca,'xticklabel',[])
g(4) = gca;
ylabel("$\phi$ ($^{\circ}$)","Interpreter","latex");
plot_set(16,2);

subplot("Position",[0.78 0.22 0.2 0.13]);
p(5) = plot(t,p_y(5,:),'Color',data_color,"LineStyle","-");
set(gca,'xticklabel',[])
g(5) = gca;
ylabel("$\theta$ ($^{\circ}$)","Interpreter","latex");
plot_set(16,2);

subplot("Position",[0.78 0.07 0.2 0.13]);
p(6) = plot(t,p_y(6,:),'Color',data_color,"LineStyle","-");
hold on
plot(t,RAD2DEG*desired_trajectory.psi_d(1,:),"LineStyle","--","Color",reference_color);
ylabel("$\psi$ ($^{\circ}$)","Interpreter","latex");
xlabel("Time (s)","Interpreter","latex");
g(6) = gca;
plot_set(16,2);


for j=1:6
    time = t(1);
    p(j).XData = time;
    p(j).YData = p_y(j,1);
    min_p_y = min(p_y(j,1:4));
    max_p_y = max(p_y(j,1:4));
    lim_y_offset = 10;
    if abs(max_p_y-min_p_y)>lim_y_offset
        lim_y_offset = abs(max_p_y-min_p_y);
    end
    g(j).YLim=[min_p_y-0.05*lim_y_offset max_p_y+0.05*lim_y_offset];

    if (time < 0.7*time_window)
        g(j).XLim=[0 time_window];
    else
        g(j).XLim=[time-0.7*time_window time+0.3*time_window];
    end
end
%% 繪製動畫準備結束

%% 起始前暫停影像
for i=1:25
     writeVideo(writerObj,getframe(video));
end
%%%

%% 開始錄製
for i = 1:n/4
ii=4*i;
%透明球
ball.XData = bal.x + data.P(1,ii);
ball.YData = bal.y + data.P(2,ii);
ball.ZData = bal.z + data.P(3,ii);

%機架
F450_dcm = angle2dcm (data.psi(ii),data.theta(ii),data.phi(ii));
F450_RT = [F450_dcm',data.P(:,ii);[0,0,0,1]];
F450_F450_poinit_new  = (F450_RT*[F450_poinit';ones(1,length(F450_poinit))])';
F450_F450_poinit_new=F450_F450_poinit_new(:,1:3);
F450.Vertices = F450_F450_poinit_new;

%槳葉

for j=1:4
    propeller_deg(j) = propeller_deg(j) + (-1)^(j)*0.3*pi;
    propeller_RT_B_to_motor{j} = [angle2dcm(propeller_deg(j),0,0)' propeller_offset{j};[0,0,0,1]];
    propeller_F450_poinit_new  = (F450_RT*propeller_RT_B_to_motor{j}*[propeller_poinit';ones(1,length(propeller_poinit))])';
    propeller_F450_poinit_new=propeller_F450_poinit_new(:,1:3);
    propeller{j}.Vertices = propeller_F450_poinit_new;
end

data_Trajectory.XData(1,i:end)=data.P(1,ii);
data_Trajectory.YData(1,i:end)=data.P(2,ii);
data_Trajectory.ZData(1,i:end)=data.P(3,ii);

% 子圖更新
time = t(ii);
for j=1:6
    
    if (time < 0.7*time_window)
        p(j).XData = t(1:ii);
        p(j).YData = p_y(j,1:ii);
        min_p_y = min(p_y(j,1:ii));
        max_p_y = max(p_y(j,1:ii));
        lim_y_offset = 10;
        if abs(max_p_y-min_p_y)>lim_y_offset
            lim_y_offset = abs(max_p_y-min_p_y);
        end
        g(j).YLim=[min_p_y-0.05*lim_y_offset max_p_y+0.05*lim_y_offset];
        g(j).XLim=[0 time_window];
    else 
        p(j).XData = t(1:ii);
        p(j).YData = p_y(j,1:ii);
        min_p_y = min(p_y(j,ceil(ii-0.7*time_window/dt):ii));
        max_p_y = max(p_y(j,ceil(ii-0.7*time_window/dt):ii));
        lim_y_offset = 10;
        if abs(max_p_y-min_p_y)>lim_y_offset
            lim_y_offset = abs(max_p_y-min_p_y);
        end
        g(j).YLim=[min_p_y-0.05*lim_y_offset max_p_y+0.05*lim_y_offset];
        if time+0.3*time_window < t(end)
            g(j).XLim=[time-0.7*time_window time+0.3*time_window];
        else
            g(j).XLim=[t(end)-time_window t(end)];
        end
    end
end

%%字體
Time_txt.String=['time: ', num2str(t(ii),'%5.2f') ,' s'];


drawnow limitrate;
writeVideo(writerObj,getframe(video));
end

%% 最後暫停畫面
ii=length(t);
%透明球
ball.XData = bal.x + data.P(1,ii);
ball.YData = bal.y + data.P(2,ii);
ball.ZData = bal.z + data.P(3,ii);

%機架
F450_dcm = angle2dcm (data.psi(ii),data.theta(ii),data.phi(ii));
F450_RT = [F450_dcm',data.P(:,ii);[0,0,0,1]];
F450_F450_poinit_new  = (F450_RT*[F450_poinit';ones(1,length(F450_poinit))])';
F450_F450_poinit_new=F450_F450_poinit_new(:,1:3);
F450.Vertices = F450_F450_poinit_new;

%槳葉
for j=1:4
    propeller_deg(j) = propeller_deg(j) + (-1)^(j)*0.3*pi;
    propeller_RT_B_to_motor{j} = [angle2dcm(propeller_deg(j),0,0)' propeller_offset{j};[0,0,0,1]];
    propeller_F450_poinit_new  = (F450_RT*propeller_RT_B_to_motor{j}*[propeller_poinit';ones(1,length(propeller_poinit))])';
    propeller_F450_poinit_new=propeller_F450_poinit_new(:,1:3);
    propeller{j}.Vertices = propeller_F450_poinit_new;
end

%%字體
Time_txt.String=['time: ', num2str(t(ii),'%5.2f') ,' s'];
drawnow limitrate;
writeVideo(writerObj,getframe(video));
for i=1:25%結束暫停影像
    writeVideo(writerObj,getframe(video));
end
%%%
close(writerObj);

function new_data=col_row_check(data)
   new_data=data;
   [m,n]=size(data);
   if(m>n)
     new_data = new_data';
   end
end
