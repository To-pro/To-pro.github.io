# 通訊方式

SD 卡可以透過 SPI 或 MMC 介面與系統進行通訊，兩者有以下的差異與優缺點：

  • SPI 介面：需要少量的引腳，且相對較容易設定，是常見的 SD 卡介面之一。SPI 介面的速度較慢，通常最高只能達到 25MHz，但是在低速應用中表現良好。由於 SPI 介面採用全雙工傳輸方式，因此可以在傳輸過程中同時進行讀取和寫入操作。

  • MMC 介面：需要較多的引腳，且相對較難設定，但是其傳輸速度較快，通常最高可達到 52MHz。MMC 介面是採用半雙工傳輸方式，因此在傳輸過程中只能進行讀取或寫入操作。

綜合來說，如果應用中需要高速讀寫資料，且有足夠的引腳可供使用，MMC 介面可能是更好的選擇；如果應用中需要少量的引腳，且速度要求不高，則可以考慮使用 SPI 介面。

# 基本讀寫

主要函數會用到如下所示:

如果是使用 SPI 通訊 (可以參考官網說明)

#include <SPI.h>
#include <SD.h>
SD.begin(SD_CS_PIN);//SPI的 CS 腳位
File file = SD.open("data.txt", FILE_WRITE);// 在 SD 卡根目錄下建立一個data.txt檔案

如果在 ESP32 使用 MMC 通訊，則 SD 要改 SD_MMC (同樣可以參考官網說明)

#include <FS.h>
#include <SD_MMC.h>
SD_MMC.begin();//MMC腳位固定
File file = SD_MMC.open("/data.txt", FILE_WRITE);// 在 SD 卡根目錄下建立一個data.txt檔案，注意ESP32目錄要加/

檔案寫入則都是相同的

file.println("Hello, SD Card!");// 寫入資料到檔案中
file.close();// 關閉檔案

# 自定義寫入

因為要擷取實驗資料我假設每次取樣有 m 個數字要記錄，直接用 println 寫入 txt 檔會有要記錄到小數點後幾位的問題，並且每次需要寫入的位元組（byte）會不一樣。因此直接將資料在記憶體中的值記錄下來就不會有上述的問題。

例如每個 double 就是 8 個 byte。那麼理論上每次只要寫入 8m8m8m byte 即可。又因為怕中間一個傳輸失誤，頭尾各加 2 個 byte 為確認 (保險) 用位元組。因此每次都寫入 8m+48m+48m+4 byte。

範例函數如下:
創建的檔案的副檔名為.bin

// 初始化 SD 卡
if (!SD_MMC.begin()) {
    Serial.println("SD 卡初始化失敗");
    return;
}
String filename_set = "data";
// 取得下一個檔案編號
int fileNumber = getNextFileNumber(filename_set);

// 新檔案名稱
String filename = "/data" + String(fileNumber, DEC) + ".bin";

// 開啟新檔案
File file = SD_MMC.open(filename, FILE_WRITE);
if (!file) {
    Serial.println("開啟檔案失敗");
    return;
}

寫入函數時使用

const char CONFIRM_BYTE1 = 0xAA;
const char CONFIRM_BYTE2 = 0x55;
template<typename T>
bool write_data_to_file(T* data,int data_length, File& filename)
{
  // 確認輸入的檔案物件是否有效
  if (!filename)
  {
      return false;
  }
  filename.write((byte*) &CONFIRM_BYTE1, sizeof(CONFIRM_BYTE1));
  filename.write((byte*) &CONFIRM_BYTE1, sizeof(CONFIRM_BYTE1));
  for (int j = 0; j < data_length; j++)
  {
    filename.write((byte*) &data[j], sizeof(data[0]));
  }
  filename.write((byte*) &CONFIRM_BYTE2, sizeof(CONFIRM_BYTE2));
  filename.write((byte*) &CONFIRM_BYTE2, sizeof(CONFIRM_BYTE2));
  filename.flush();
  return true;
}

另外如果想檢查是否已存在檔案則可以參考下面所示

int getNextFileNumber(String fileName ) {
  int fileNumber = 1; // 起始檔案編號
  while (SD_MMC.exists("/" + fileName + String(fileNumber, DEC) + ".bin")) {
    fileNumber++; // 檢查下一個檔案編號
  }
//   SD_MMC.mkdir("/data" + String(fileNumber, DEC) + ".bin");
  return fileNumber;
}

# MATLAB 讀取方式

因為紀錄的.bin 檔還需要解碼，這裡提供 MATLAB 程式讀取並繪圖

clc;clear;close all;
addpath 'Z:\My Drive\MATLAB\plot_function'

fid = fopen('data1.bin', 'r');

%定義開頭與結尾的 byte
start_byte = 0xAA;
end_byte = 0x55;

%定義每筆資料長度與資料的型態(float or double)
data_length = 22;
data_front = 'double';

% 讀取數據
data = [];
while ~feof(fid)
    % 確認開頭的 byte
    byte1_1 = fread(fid, 1, 'uint8');
    if feof(fid)
        disp('读取完毕');
        break
    end
    byte1_2 = fread(fid, 1, 'uint8');
    while ((byte1_1 ~= start_byte) &&(byte1_2 ~= start_byte) && (~feof(fid)))
        byte1 = fread(fid, 1, 'uint8');
    end
    if ~feof(fid)
        % 讀取data_length個double
        tmp = fread(fid, data_length, data_front);
        % 確認結尾 byte
        byte2_1 = fread(fid, 1, 'uint8');
        byte2_2 = fread(fid, 1, 'uint8');
        if ((byte2_1 == end_byte)&&(byte2_2 == end_byte))
            % 加入資料中
            data = [data; tmp'];
        end
    end
end
% 關閉文件
fclose(fid);

duration = data(2:end,1);
t = data(:,2);
x = data(:,3:end);
fig(1)=figure("Name","diff_t");
plot(diff(t),'-b');
ylabel("Diff t")
xlim tight
ylim padded
plot_set(14,2);

fig(2)=figure("Name","duration");
plot(duration,'-b');
ylabel("duration(ms)")
xlim tight
ylim padded
plot_set(14,2);

fig(3)=figure("Name","duration_Hz");
plot(ones(size(duration))./duration*1000,'-b');
ylabel("duration(Hz)")
xlim tight
ylim padded
plot_set(14,2);
min(ones(size(duration))./duration*1000)

fig(4)=figure("Name","data_x");
plot(t,x(:,1),'-b');
ylabel("sin(t)")
xlim tight
ylim padded
plot_set(14,2);

# 函示庫懶人包 (連結)

這裡將程式碼包成函示庫方便以後調用:

  • 注意事項

    1. 預設是 MMC 通訊方式，若是用 SPI 則需在引入函數庫前定義 SD_SPI

    2. SD_save 為結構變數，請在使用任何本函數庫中的函數前設定好

    3. My_SD_init (“data”); 引數為創建的檔案名稱，若與已存在重名則在最後面加上數字

    4. My_SD_init_info_save (); 可加可不加，需再 My_SD_init 後。會再創一個同名的 txt 檔紀錄寫入的 SD 卡的相關資訊 (確認 byte、資料長度和資料形式)

  • 主程式範例:
    資料為 sin (t)，t 為時間 (0~t_end，間隔 dt)

  • MATLAB (要有用 My_SD_init_info_save () 紀錄寫入的 SD 卡的相關資訊的 txt 才可以使用)
